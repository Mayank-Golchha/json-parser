<JSON>     ::= <value>
<value>    ::= <object> | <array> | <boolean> | <string> | <number> | <null>
<array>    ::= "[" [ <value> *(', ' <value>) ] "]"
<object>   ::= "{" [ <property> *(', ' <property>) ] "}"
<property> ::= <string> ":" <value>
<children-property> ::= <comma> <property> | NONE
<children-value> ::= <comma> <value> | NONE
